String.prototype

String.$trim()

String.$trim() will remove the spaces at the beginning and end.

$C(function(){
    var str = "     hello world    ";
    console.log(str.$removeSpace()); //"hello world"
})

String.$removeSpace()

String.$removeSpace() will remove the spaces between the words.

$C(function(){
    var str = "hello world ! ! !";
    console.log(str.$removeSpace()); //"helloworld!!!"
})
